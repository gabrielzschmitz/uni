\section{Problemas}

\question{
  Dados os vetores $\vec{u} = 3\vec{i} - \vec{j} - 2\vec{k}$, $\vec{v} =
  2\vec{i} + 4\vec{j} - \vec{k}$ e $\vec{w} = -\vec{i} + \vec{k}$, determinar:
}
\begin{itemize}
  \item \subquestion{$|\vec{u} \times \vec{u}|$}
    \answer{
      \begin{align*}
        \vec{u} \times \vec{u} = \vec{0}
      \end{align*}
    }
  \item \subquestion{$(2\vec{v}) \times (3\vec{v})$}
    \answer{
      \begin{align*}
        2\vec{v} \times 3\vec{v} &= \vec{0} \\
      \end{align*}
    }
  \item \subquestion{$(\vec{u} \times \vec{w}) + (\vec{w} \times \vec{u})$}
    \answer{
      \begin{align*}
        (\vec{u} \times \vec{w}) + (\vec{w} \times \vec{u}) &= \\
        (\vec{u} \times \vec{w}) - (\vec{u} \times \vec{w}) &= \vec{0} \\
      \end{align*}
    }
  \item \subquestion{$(\vec{u} \times \vec{v}) \times (\vec{v} \times \vec{u})$}
    \answer{
      \begin{align*}
        (\vec{u} \times \vec{v}) \times (\vec{v} \times \vec{u}) &= \\
        &= -((\vec{u} \times \vec{v}) \times (\vec{u} \times \vec{v})) \\
        &= -\vec{0} \\
        &= \vec{0} \\
      \end{align*}
    }
  \item \subquestion{$(\vec{u} - \vec{v}) \times \vec{w}$}
    \answer{
      \begin{align*}
        \vec{u} - \vec{v} &= \vec{i} - 5\vec{j} - \vec{k} \\
        (\vec{u} - \vec{v}) \times \vec{w} &= 
        \begin{vmatrix}
          \vec{i} & \vec{j} & \vec{k} & \vec{i} & \vec{j}\\
          1 & -5 & -1 & 1 & -5 \\
          -1 & 0 & 1 & -1 & 0 \\
        \end{vmatrix} \\
        &= -5\vec{i} - 5\vec{k} \\
        &= (-5, 0, -5) \\
      \end{align*}
    }
\end{itemize}

\question{
  Dados os pontos $A(2, 1, -1)$, $B(3, 0, 1)$ e $C(2, -1, -3)$, determinar o
  ponto $D$ tal que $\overrightarrow{AD} = \overrightarrow{BC} \times
  \overrightarrow{AC}$.
}
\answer{
  \begin{aligned}
    \overrightarrow{BC} &= C - B = (2-3, -1-0, -3-1) = (-1, -1, -4) \\
    \overrightarrow{AC} &= C - A = (2-2, -1-1, -3-(-1)) = (0, -2, -2) \\
    \overrightarrow{BC} \times \overrightarrow{AC} &= \begin{vmatrix} \\
      \vec{i} & \vec{j} & \vec{k} \\
      -1 & -1 & -4 \\
      0 & -2 & -2 \\
    \end{vmatrix} \\
    &= \vec{i}((-1)(-2)-(-4)(-2)) - \vec{j}((-1)(-2)-(-4)(0)) + \vec{k}((-1)(-2)-(-1)(0)) \\
    &= \vec{i}(2-8) - \vec{j}(2-0) + \vec{k}(2-0) \\
    &= (-6, -2, 2) \\
    \overrightarrow{AD} &= D - A = (-6, -2, 2) \\
    D &= A + \overrightarrow{AD} = (2, 1, -1) + (-6, -2, 2) \\
    &= (-4, -1, 1) \\
  \end{aligned}
}

\question{
  Determinar o vetor $\vec{x}$ tal que: $\vec{x} \cdot (1, 4, -3) = -7$ e
  $\vec{x} \times (4, -2, 1) = (3, 5, -2)$
}
\answer{
  \begin{aligned}
    \vec{x} &= (x_1, x_2, x_3) \\
    x_1 + 4x_2 - 3x_3 &= -7 \\
    x_2 + 2x_3 &= 3 \\
    x_1 - 4x_3 &= -5 \\
    2x_1 + 4x_2 &= 2 \\
    x_2 &= 3 - 2x_3 \\
    x_1 &= -5 + 4x_3 \\
    2(-5 + 4x_3) + 4(3 - 2x_3) &= 2 \\
    -10 + 8x_3 + 12 - 8x_3 &= 2 \\
    2 &= 2 \quad \text{(ok)} \\
    x_3 &= 2 \\
    x_2 &= 3 - 2 \cdot 2 = -1 \\
    x_1 &= -5 + 4 \cdot 2 = 3 \\
    \vec{x} &= (3,\ -1,\ 2) \\
  \end{aligned}
}

\question{
  Resolver os sistemas: \\
}
\subquestion{
  \[
    \begin{cases}
      \vec{x} \times \vec{j} = \vec{k} \\
      \vec{x} \cdot (4\vec{i} - 2\vec{j} + \vec{k}) = 10
    \end{cases}
  \]
}
\answer{
  \text{1º sistema:} \\
  \vec{x} &= (x_1, x_2, x_3) \\
  \vec{x} \times \vec{j} &= \vec{k} \Rightarrow 
  \begin{vmatrix}
    \vec{i} & \vec{j} & \vec{k} \\
    x_1 & x_2 & x_3 \\
    0 & 1 & 0 \\
  \end{vmatrix}
  = (-x_3, 0, x_1) \\
  (-x_3, 0, x_1) &= (0, 0, 1) \Rightarrow 
  \begin{cases}
    -x_3 = 0 \Rightarrow x_3 = 0 \\
    x_1 = 1
  \end{cases} \\
  \vec{x} \cdot (4, -2, 1) &= 10 \Rightarrow 
  4x_1 - 2x_2 + x_3 = 10 \Rightarrow 4(1) - 2x_2 + 0 = 10 \\
  -2x_2 &= 6 \Rightarrow x_2 = -3 \\
  \vec{x} &= (1,\ -3,\ 0) \\
}
\subquestion{
  \[
    \begin{cases}
      \vec{x} \times (2\vec{i} - \vec{j} + 3\vec{k}) = \vec{0} \\
      \vec{x} \cdot (\vec{i} + 2\vec{j} - 2\vec{k}) = 12
    \end{cases}
  \]
}
\answer{
  \text{2º sistema:} \\
  \vec{x} &= (x_1, x_2, x_3) \\
  \vec{x} \times (2, -1, 3) = \vec{0} \Rightarrow \text{vetores paralelos} \Rightarrow 
  \vec{x} = \lambda(2,\ -1,\ 3) \\
  \vec{x} \cdot (1, 2, -2) = 12 \Rightarrow 
  \lambda(2 \cdot 1 + (-1) \cdot 2 + 3 \cdot (-2)) = 12 \\
  \lambda(2 - 2 - 6) = 12 \Rightarrow -6\lambda = 12 \Rightarrow \lambda = -2 \\
  \vec{x} &= (-4,\ 2,\ -6)
}

\question{
  Dados os vetores $\vec{u} = (3,\ 1,\ 1)$, $\vec{v} = (-4,\ 1,\ 3)$ e $\vec{w}
  = (1,\ 2,\ 0)$, determinar $\vec{x}$ de modo que $\vec{x} \perp \vec{w}$ e
  $\vec{x} = \vec{u} \times \vec{v}$.
}
\answer{
  \begin{aligned}
    \vec{x} &= \vec{u} \times \vec{v} \\
    &= 
    \begin{vmatrix}
      \vec{i} & \vec{j} & \vec{k} \\
      3 & 1 & 1 \\
      -4 & 1 & 3 \\
    \end{vmatrix}
    = (2,\ -13,\ 7) \\
    \vec{x} \cdot \vec{w} &= 2 \cdot 1 + (-13) \cdot 2 + 7 \cdot 0 = -24 \neq 0 \\
    \text{Logo, } &\vec{x} \not\perp \vec{w} \\
    \text{Além disso, } &\vec{u} \cdot \vec{v} = 3 \cdot (-4) + 1 \cdot 1 + 1
    \cdot 3 = -12 + 1 + 3 = -8 \neq 0 \\
    \text{Portanto, } &\text{não existe } \vec{x} \text{ tal que } \vec{x} =
    \vec{u} \times \vec{v} \text{ e } \vec{x} \perp \vec{w}, \text{ pois }
    \vec{u} \not\perp \vec{v}.
  \end{aligned}
}
